// npm i rate-limiter-flexible ioredis
const { RateLimiterRedis } = require('rate-limiter-flexible');
const Redis = require('ioredis');

const redis = new Redis(process.env.REDIS_URL);

const searchLimiter = new RateLimiterRedis({
  storeClient: redis,
  keyPrefix: 'rl:search',
  points: 10,        // requests
  duration: 1,        // per second
  blockDuration: 10,  // block for 10s once exceeded
});

async function searchRateLimit(req, res, next) {
  const key = req.user?.id || req.ip; // prefer authed user id if available
  try {
    await searchLimiter.consume(key);
    next();
  } catch (rejRes) {
    res.set('Retry-After', String(Math.ceil(rejRes.msBeforeNext / 1000)));
    res.status(429).json({ error: 'Too many requests' });
  }
}

app.get('/search', searchRateLimit, searchHandler);
