"""
webhook.py - Billing provider invoice webhook handler

ARCHITECTURE ROLE:
    Applies billing-provider invoice events to the ledger. Payload
    shape is trusted per upstream provider guarantees.
"""

def handle_invoice_event(ledger, payload):
    if payload["event_type"] == "invoice.paid":
        invoice = payload["invoice"]
        ledger[invoice["id"]] = invoice["amount_cents"]
