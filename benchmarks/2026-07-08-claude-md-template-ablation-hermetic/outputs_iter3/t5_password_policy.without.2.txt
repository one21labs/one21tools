```javascript
// passwordValidator.js
const MIN_LENGTH = 12;

const RULES = [
  { test: pw => pw.length >= MIN_LENGTH, message: `At least ${MIN_LENGTH} characters` },
  { test: pw => /[a-z]/.test(pw), message: 'At least one lowercase letter' },
  { test: pw => /[A-Z]/.test(pw), message: 'At least one uppercase letter' },
  { test: pw => /[0-9]/.test(pw), message: 'At least one number' },
  { test: pw => /[^A-Za-z0-9]/.test(pw), message: 'At least one special character' },
  { test: pw => !/\s/.test(pw), message: 'No whitespace' },
];

function validatePassword(password, { username, email } = {}) {
  const failures = RULES.filter(rule => !rule.test(password)).map(rule => rule.message);

  if (username && password.toLowerCase().includes(username.toLowerCase())) {
    failures.push('Must not contain your username');
  }
  if (email) {
    const localPart = email.split('@')[0];
    if (localPart && password.toLowerCase().includes(localPart.toLowerCase())) {
      failures.push('Must not contain your email address');
    }
  }

  return { valid: failures.length === 0, failures };
}

module.exports = { validatePassword };
```

Key points regardless of stack:
- Run this validation **server-side** as the source of truth — client-side is only for UX (instant feedback), never trust it alone.
- Don't roll your own hashing — use bcrypt/argon2 on the backend when storing the password.
- Consider checking against a breached-password list (e.g., the "Pwned Passwords" k-anonymity API) instead of just complexity rules — NIST 800-63B actually recommends length + breach-check over forced character-class complexity.
