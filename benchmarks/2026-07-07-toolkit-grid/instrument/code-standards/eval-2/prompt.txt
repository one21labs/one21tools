import logging

log = logging.getLogger("payments")

FEE_RATE = 0.02

def process_refund(orders, payments_api, order_id, amount):
    # look up the order
    order = orders[order_id]
    order.status = "refunded"
    order.refunded_amount = amount
    try:
        result = payments_api.refund(order.payment_ref, amount)
    except Exception:
        log.error("refund error")
        return False
    if result.get("status") != "ok":
        log.warning("refund not ok")
    order.fee = amount * FEE_RATE
    return True
