## scripts/fetch_users.py

​```python
import logging
import time

import requests

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(name)s: %(message)s")
logger = logging.getLogger(__name__)

BASE = "https://api.newhost.io/v2"

MAX_RETRIES = 3
RETRY_DELAY_SECONDS = 2


def fetch_users():
    url = BASE + "/users"
    for attempt in range(1, MAX_RETRIES + 1):
        logger.info("Fetching users from %s (attempt %d/%d)", url, attempt, MAX_RETRIES)
        try:
            response = requests.get(url)
            response.raise_for_status()
            logger.info("Fetched users successfully (status %d)", response.status_code)
            return response.json()
        except requests.RequestException as exc:
            logger.error("Request to %s failed on attempt %d/%d: %s", url, attempt, MAX_RETRIES, exc)
            if attempt == MAX_RETRIES:
                logger.error("All %d attempts failed for %s", MAX_RETRIES, url)
                raise
            time.sleep(RETRY_DELAY_SECONDS)
