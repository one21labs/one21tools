Here are both modules with production-grade logging added. Design summary: every log line carries a `run_id` (unique per invocation) and `record_id` (`path:line_number`, stable across re-runs of the same file) so you can grep by `record_id` to follow one record through ingest → enrich → save, or by `run_id` to scope to a single run. Per-record success is logged at `DEBUG` (enable on demand to trace a specific record without flooding production logs); anomalies (parse errors, unknown region, save failures, unexpected exceptions) are logged at `WARNING`/`ERROR` with full context and `exc_info=True`, so they're visible at the default `INFO` level and diagnosable without reproducing the run. A single bad record no longer kills the whole batch — it's caught, logged, counted, and processing continues; a run-level summary (`read/saved/skipped/failed`) is emitted at the end.

**ingest.py**
```python
import json
import logging
import sys
import uuid

from enrich import enrich_record

logger = logging.getLogger(__name__)


class _ContextFilter(logging.Filter):
    """Defaults run_id/record_id so the format string never KeyErrors on
    log calls (e.g. run-level summaries) that don't pass them explicitly."""

    def filter(self, record):
        record.run_id = getattr(record, "run_id", "-")
        record.record_id = getattr(record, "record_id", "-")
        return True


def configure_logging(level=logging.INFO):
    handler = logging.StreamHandler()
    handler.addFilter(_ContextFilter())
    handler.setFormatter(logging.Formatter(
        "%(asctime)s %(levelname)-8s %(name)s "
        "run_id=%(run_id)s record_id=%(record_id)s %(message)s"
    ))
    root = logging.getLogger()
    root.handlers = [handler]
    root.setLevel(level)


def run(path):
    run_id = uuid.uuid4().hex[:12]
    ctx = {"run_id": run_id, "record_id": "-"}
    logger.info("run started path=%s", path, extra=ctx)

    stats = {"read": 0, "saved": 0, "skipped": 0, "failed": 0}

    try:
        f = open(path)
    except OSError:
        logger.error("cannot open input file path=%s", path, extra=ctx, exc_info=True)
        raise

    with f:
        for line_no, line in enumerate(f, start=1):
            record_id = f"{path}:{line_no}"
            rctx = {"run_id": run_id, "record_id": record_id}
            stats["read"] += 1
            raw = line.rstrip("\n")

            try:
                rec = json.loads(raw)
            except json.JSONDecodeError as e:
                stats["failed"] += 1
                logger.error("failed to parse record: %s raw=%r", e, raw[:200], extra=rctx)
                continue

            logger.debug("record parsed", extra=rctx)

            try:
                out = enrich_record(rec, run_id=run_id, record_id=record_id)
            except Exception:
                stats["failed"] += 1
                logger.error("enrichment raised an unexpected error", extra=rctx, exc_info=True)
                continue

            if out is None:
                stats["skipped"] += 1
                logger.warning("record skipped by enrichment", extra=rctx)
                continue

            try:
                save(out)
            except OSError:
                stats["failed"] += 1
                logger.error("failed to save record", extra=rctx, exc_info=True)
                continue

            stats["saved"] += 1
            logger.debug("record saved", extra=rctx)

    logger.info(
        "run finished read=%d saved=%d skipped=%d failed=%d",
        stats["read"], stats["saved"], stats["skipped"], stats["failed"],
        extra=ctx,
    )


def save(rec):
    with open("out.jsonl", "a") as f:
        f.write(json.dumps(rec) + "\n")


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("usage: python ingest.py <path>", file=sys.stderr)
        sys.exit(1)
    configure_logging(level=logging.INFO)
    run(sys.argv[1])
```

**enrich.py**
```python
import logging

logger = logging.getLogger(__name__)

GEO = {"us": "US-EAST", "eu": "EU-WEST"}


def enrich_record(rec, run_id="-", record_id="-"):
    ctx = {"run_id": run_id, "record_id": record_id}

    region_key = rec.get("region")
    if region_key is None:
        logger.warning("record missing 'region' field, skipping", extra=ctx)
        return None

    region = GEO.get(region_key)
    if region is None:
        logger.warning("unrecognized region=%r, skipping", region_key, extra=ctx)
        return None

    rec["zone"] = region
    logger.debug("record enriched region=%s zone=%s", region_key, region, extra=ctx)
    return rec
```

Key behavior changes from the original: `rec["region"]` (which would `KeyError`-crash the whole run on a malformed record) is now `rec.get("region")` with an explicit, logged skip; malformed JSON lines and I/O failures on save are caught and logged instead of aborting the batch; and `enrich_record` gained two optional trailing kwargs (`run_id`, `record_id`) purely for log correlation — behavior and return value are unchanged for callers that don't pass them.
