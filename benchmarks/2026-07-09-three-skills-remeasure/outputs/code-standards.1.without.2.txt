```python
import json
import time
import urllib.request

# old loader, kept for reference
# def load(f):
#     return json.loads(open(f).read())

def ld(p):
    try:
        d = json.load(open(p))
    except:
        d = {}
    # get the retries
    r = d.get("retries", 5)
    to = d.get("timeout", 30)
    bo = d.get("backoff_seconds", 0)
    return r, to, bo

def snd(u, pl, r, to, bo=0):
    for i in range(r):
        try:
            req = urllib.request.Request(u, data=pl)
            return urllib.request.urlopen(req, timeout=to).read()
        except:
            pass
        if bo and i < r - 1:
            time.sleep(bo)
    return None
```

Changes: `ld()` now also reads `backoff_seconds` from the config (defaulting to `0`, i.e. no wait) and returns it alongside `r`/`to`; `snd()` takes a new `bo` parameter and sleeps that many seconds between attempts (skipped after the final attempt, and skipped entirely if `bo` is falsy). Callers need updating from `r, to = ld(p)` to `r, to, bo = ld(p)` and `snd(u, pl, r, to)` to `snd(u, pl, r, to, bo)`.
