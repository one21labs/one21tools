```bash
#!/usr/bin/env bash
# ci/check-file-length.sh
# Fails the build if any file under src/ exceeds 500 lines.

set -euo pipefail

MAX_LINES=500
SRC_DIR="src"
FAILED=0

while IFS= read -r -d '' file; do
  lines=$(wc -l < "$file")
  if [ "$lines" -gt "$MAX_LINES" ]; then
    echo "FAIL: $file has $lines lines (max $MAX_LINES)"
    FAILED=1
  fi
done < <(find "$SRC_DIR" -type f -print0)

if [ "$FAILED" -eq 1 ]; then
  echo "One or more files exceed $MAX_LINES lines."
  exit 1
fi

echo "All files under $SRC_DIR are within $MAX_LINES lines."
```

Usage: `chmod +x ci/check-file-length.sh` and call it as a CI step (`./ci/check-file-length.sh`). If you only want specific extensions checked (e.g. `.ts`, `.py`), tell me and I'll add a `-name` filter to the `find` call.
