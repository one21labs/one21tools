**Node.js / Express + Redis example:**

```javascript
// cache.js
const redis = require('redis');

const client = redis.createClient({ url: process.env.REDIS_URL });
client.connect();

const DEFAULT_TTL_SECONDS = 300;

async function getOrSetCache(key, ttlSeconds, fetchFn) {
  const cached = await client.get(key);
  if (cached) return JSON.parse(cached);

  const fresh = await fetchFn();
  await client.set(key, JSON.stringify(fresh), { EX: ttlSeconds });
  return fresh;
}

module.exports = { client, getOrSetCache, DEFAULT_TTL_SECONDS };
```

```javascript
// routes/product.js
const { getOrSetCache, DEFAULT_TTL_SECONDS } = require('../cache');
const { getProductById } = require('../services/productService');

router.get('/products/:id', async (req, res, next) => {
  try {
    const { id } = req.params;
    const cacheKey = `product:${id}`;

    const product = await getOrSetCache(cacheKey, DEFAULT_TTL_SECONDS, () =>
      getProductById(id)
    );

    if (!product) return res.status(404).json({ error: 'Not found' });
    res.json(product);
  } catch (err) {
    next(err);
  }
});
```

Key decisions to confirm for your case:

- **Invalidation** — cache-aside with explicit `del` on writes shown above; alternatives are TTL-only or event-driven invalidation.
