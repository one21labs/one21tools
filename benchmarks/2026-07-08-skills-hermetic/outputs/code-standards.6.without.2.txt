```python
import sqlite3

def add_user(db, name, email):
    try:
        db.execute("insert into users (name, email) values (?, ?)", (name, email))
        db.commit()
        print("ok")
        return 1
    except:
        return 0

def recent_users(db):
    try:
        return db.execute("select * from users order by created desc limit 50").fetchall()
    except:
        return []

def search_users(db, q):
    try:
        # search the users
        return db.execute("select * from users where name like ? limit 50", ("%" + q + "%",)).fetchall()
    except:
        return []

def find_users_by_domain(db, domain):
    try:
        return db.execute("select * from users where email like ? limit 50", ("%@" + domain,)).fetchall()
    except:
        return []
```

Followed the existing pattern exactly: try/except returning `[]` on failure, parameterized query (so a domain like `%foo` can't break out of the LIKE), and the same 50-row cap as `recent_users`/`search_users`. One note — `%` or `_` in `domain` will act as SQL LIKE wildcards since none of the existing queries escape them either; that's consistent with `search_users`' existing behavior, so I left it as-is rather than introducing a new escaping convention on its own.
