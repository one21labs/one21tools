```python
import json
import time
import urllib.request


def ld(p):
    try:
        d = json.load(open(p))
    except:
        d = {}
    # get the retries
    r = d.get("retries", 5)
    to = d.get("timeout", 30)
    bo = d.get("backoff_seconds", 0)
    return r, to, bo


def snd(u, pl, r, to, bo=0):
    for i in range(r):
        try:
            req = urllib.request.Request(u, data=pl)
            return urllib.request.urlopen(req, timeout=to).read()
        except:
            pass
        if bo and i < r - 1:
            time.sleep(bo)
    return None
```

- Sleep is skipped after the final attempt so a failed call doesn't wait needlessly before returning `None`.
