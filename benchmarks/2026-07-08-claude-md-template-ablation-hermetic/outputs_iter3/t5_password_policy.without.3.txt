In the meantime, here's a solid, framework-agnostic validator you can adapt (follows NIST 800-63B guidance — length matters more than forced complexity, and blocking known-bad passwords beats arbitrary character-class rules):

```javascript
const MIN_LENGTH = 12;
const MAX_LENGTH = 128;
const COMMON_PASSWORDS = new Set([
  'password', 'password123', '123456789', 'qwertyuiop', 'letmein123',
  // load a larger breached-password list (e.g., Have I Been Pwned's list) in production
]);

function validatePasswordStrength(password, { username = '', email = '' } = {}) {
  const errors = [];

  if (typeof password !== 'string' || password.length === 0) {
    return { valid: false, errors: ['Password is required'] };
  }
  if (password.length < MIN_LENGTH) {
    errors.push(`Password must be at least ${MIN_LENGTH} characters`);
  }
  if (password.length > MAX_LENGTH) {
    errors.push(`Password must be at most ${MAX_LENGTH} characters`);
  }
  if (COMMON_PASSWORDS.has(password.toLowerCase())) {
    errors.push('Password is too common — choose something less predictable');
  }
  const lowered = password.toLowerCase();
  if (username && lowered.includes(username.toLowerCase())) {
    errors.push('Password must not contain your username');
  }
  if (email) {
    const localPart = email.split('@')[0].toLowerCase();
    if (localPart.length > 2 && lowered.includes(localPart)) {
      errors.push('Password must not contain your email address');
    }
  }
  // Optional character-class checks — NIST says these add little real security
  // benefit and hurt usability, so keep them light or skip them entirely.
  if (!/[a-z]/.test(password) || !/[A-Z]/.test(password) || !/[0-9]/.test(password)) {
    errors.push('Password must include uppercase, lowercase, and a number');
  }

  return { valid: errors.length === 0, errors };
}

module.exports = { validatePasswordStrength };
```

Server-side validation is the one that actually matters for security — always re-run this check on submit, not just in the browser. Client-side, you'd wire it into the form's `onSubmit`/`onChange` to show live feedback.
