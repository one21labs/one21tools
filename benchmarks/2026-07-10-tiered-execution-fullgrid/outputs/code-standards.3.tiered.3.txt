```python
# ingest.py
import json
import logging
from enrich import enrich_record

logger = logging.getLogger(__name__)

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s %(name)s %(levelname)s %(message)s'
)

def run(path):
    total_read = 0
    total_saved = 0
    total_dropped = 0
    total_errors = 0
    
    logger.info(f"event=run_started path={path}")
    
    try:
        with open(path) as f:
            for line_no, line in enumerate(f, 1):
                total_read += 1
                record_id = f"{path}:{line_no}"
                
                try:
                    rec = json.loads(line)
                except json.JSONDecodeError as e:
                    total_errors += 1
                    logger.warning(f"record_id={record_id} event=record_dropped reason=parse_error error_type={type(e).__name__}")
                    continue
                
                rec["_record_id"] = record_id
                
                logger.debug(f"record_id={record_id} event=record_received")
                
                try:
                    out = enrich_record(rec)
                except Exception as e:
                    total_errors += 1
                    logger.error(f"record_id={record_id} event=enrich_failed error_type={type(e).__name__}", exc_info=True)
                    continue
                
                if out is None:
                    total_dropped += 1
                    continue
                
                try:
                    out_to_save = {k: v for k, v in out.items() if k != "_record_id"}
                    save(out_to_save)
                    total_saved += 1
                    logger.info(f"record_id={record_id} event=record_saved")
                except Exception as e:
                    total_errors += 1
                    logger.error(f"record_id={record_id} event=save_failed error_type={type(e).__name__}", exc_info=True)
                    continue
    
    except Exception as e:
        logger.error(f"event=run_failed error_type={type(e).__name__}", exc_info=True)
        raise
    
    logger.info(f"event=run_completed path={path} total_read={total_read} total_saved={total_saved} total_dropped={total_dropped} total_errors={total_errors}")

def save(rec):
    with open("out.jsonl", "a") as f:
        f.write(json.dumps(rec) + "\n")
```

```python
# enrich.py
import logging

logger = logging.getLogger(__name__)

GEO = {"us": "US-EAST", "eu": "EU-WEST"}

def enrich_record(rec):
    record_id = rec.get("_record_id", "unknown")
    
    try:
        region = rec["region"]
    except KeyError:
        logger.warning(f"record_id={record_id} event=record_dropped reason=missing_region_field")
        return None
    
    mapped_region = GEO.get(region)
    if mapped_region is None:
        logger.warning(f"record_id={record_id} event=record_dropped reason=unknown_region region={region}")
        return None
    
    rec["zone"] = mapped_region
    logger.info(f"record_id={record_id} event=record_enriched zone={mapped_region}")
    return rec
```
