```python
import sqlite3

def add_user(db, name, email):
    try:
        db.execute("insert into users (name, email) values (?, ?)", (name, email))
        db.commit()
        print("ok")
        return 1
    except:
        return 0

def recent_users(db):
    try:
        return db.execute("select * from users order by created desc limit 50").fetchall()
    except:
        return []

def search_users(db, q):
    try:
        # search the users
        return db.execute("select * from users where name like ? limit 50", ("%" + q + "%",)).fetchall()
    except:
        return []

def find_users_by_domain(db, domain):
    try:
        return db.execute("select * from users where email like ? limit 50", ("%@" + domain,)).fetchall()
    except:
        return []
```

Kept the same try/except-with-bare-except, `.fetchall()`, and `limit 50` pattern as the other read queries. The `%@domain` value is still passed as a bound parameter (not string-interpolated into the SQL), so it matches `search_users`'s existing safe parameterization style rather than introducing injection risk.
