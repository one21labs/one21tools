```python
import json
import time
import urllib.request

def ld(p):
    try:
        d = json.load(open(p))
    except:
        d = {}
    r = d.get("retries", 5)
    to = d.get("timeout", 30)
    backoff = d.get("backoff_seconds", 0)
    return r, to, backoff

def snd(u, pl, r, to, backoff=0):
    for i in range(r):
        try:
            req = urllib.request.Request(u, data=pl)
            return urllib.request.urlopen(req, timeout=to).read()
        except:
            if i < r - 1 and backoff:
                time.sleep(backoff)
    return None
```

Changes:
- `ld` now also reads `backoff_seconds` from the config (defaults to `0`, i.e. no wait) and returns it alongside `retries`/`timeout`.
- `snd` takes a `backoff` param and sleeps that many seconds between attempts — only between retries, not after the final failed attempt.
