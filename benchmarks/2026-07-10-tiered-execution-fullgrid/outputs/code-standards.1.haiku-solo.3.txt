```python
import json
import time
import urllib.request

def ld(p):
    try:
        d = json.load(open(p))
    except:
        d = {}
    # get the retries, timeout, and backoff_seconds
    r = d.get("retries", 5)
    to = d.get("timeout", 30)
    bs = d.get("backoff_seconds", 1)
    return r, to, bs

def snd(u, pl, r, to, bs):
    for i in range(r):
        try:
            req = urllib.request.Request(u, data=pl)
            return urllib.request.urlopen(req, timeout=to).read()
        except:
            if i < r - 1:  # don't sleep after the last attempt
                time.sleep(bs)
    return None
```

Changes:
- `ld()` now reads `backoff_seconds` from config (defaults to 1 second) and returns it as a third value
- `snd()` accepts `bs` parameter and sleeps between retry attempts (but not after the final one)
