```python
import logging
import time
import requests

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# UPDATED (from request): Changed URL
BASE = "https://api.newhost.io/v2"

# NEW: Minimal retry wrapper around the original requests.get(BASE + "/users")
max_retries = 3
for attempt in range(max_retries):
    try:
        response = requests.get(BASE + "/users")
        logger.info(f"Successfully fetched users from {BASE + '/users'}")
        # [existing response-handling code continues here]
        break
    except requests.RequestException as e:
        logger.error(f"Attempt {attempt + 1}/{max_retries} failed: {e}")
        if attempt < max_retries - 1:
            time.sleep(1)
        else:
            raise
```
