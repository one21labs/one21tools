## scripts/fetch_users.py

```python
import logging
import time

import requests

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(name)s: %(message)s",
)
logger = logging.getLogger("fetch_users")

BASE = "https://api.newhost.io/v2"

MAX_RETRIES = 3
RETRY_BACKOFF_SECONDS = 2


def fetch_users():
    url = BASE + "/users"
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            logger.info("Requesting %s (attempt %d/%d)", url, attempt, MAX_RETRIES)
            resp = requests.get(url, timeout=10)
            resp.raise_for_status()
            logger.info("Fetched users OK (%d bytes)", len(resp.content))
            return resp.json()
        except requests.RequestException as exc:
            logger.warning("Attempt %d failed: %s", attempt, exc)
            if attempt == MAX_RETRIES:
                logger.error("Giving up after %d attempts", MAX_RETRIES)
                raise
            time.sleep(RETRY_BACKOFF_SECONDS * attempt)


if __name__ == "__main__":
    fetch_users()
```
