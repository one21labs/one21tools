```python
import sqlite3

def add_user(db, name, email):
    try:
        db.execute("insert into users (name, email) values (?, ?)", (name, email))
        db.commit()
        print("ok")
        return 1
    except:
        return 0

def recent_users(db):
    try:
        return db.execute("select * from users order by created desc limit 50").fetchall()
    except:
        return []

def search_users(db, q):
    try:
        # search the users
        return db.execute("select * from users where name like ? limit 50", ("%" + q + "%",)).fetchall()
    except:
        return []

def find_users_by_domain(db, domain):
    try:
        return db.execute("select * from users where email like ? limit 50", ("%@" + domain,)).fetchall()
    except:
        return []
```

Followed the existing pattern exactly: bare `try/except` returning `[]` on failure, parameterized query, and the same 50-row cap as `recent_users`/`search_users`. The domain match uses `like ? ` with `"%@" + domain` so it matches emails ending in `@<domain>` — note this is a straight suffix match, so a domain of `evil.com` would also match `notevil.com` if it lacked the `@`; the leading `@` in the pattern is what keeps it anchored to the actual domain part of the email.
