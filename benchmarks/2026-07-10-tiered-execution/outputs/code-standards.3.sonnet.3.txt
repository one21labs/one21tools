I don't have filesystem access in this session, so here are the two modules as text — save them as `ingest.py` and `enrich.py`.

Design: every record gets a `record_id` (its own `id` field if present, otherwise `path:line_number`) that's threaded explicitly from `ingest.py` into `enrich_record()`, so a single record's path is greppable across both modules. Each run also gets a `run_id` to disambiguate concurrent/successive runs in a shared log stream. Logs are structured JSON (one object per line) so they're machine-parseable in production. Per-record processing is wrapped so one bad line (bad JSON, missing field, write failure) is logged with full context and the run continues instead of crashing; a summary with success/skip/fail counts is logged at run end.

**ingest.py**
```python
import json
import logging
import time
import uuid

from enrich import enrich_record

logger = logging.getLogger(__name__)


class JsonFormatter(logging.Formatter):
    def format(self, record):
        payload = {
            "time": self.formatTime(record, "%Y-%m-%dT%H:%M:%S"),
            "level": record.levelname,
            "module": record.name,
        }
        if isinstance(record.msg, dict):
            payload.update(record.msg)
        else:
            payload["msg"] = record.getMessage()
        if record.exc_info:
            payload["exc_info"] = self.formatException(record.exc_info)
        return json.dumps(payload)


def _configure_logging():
    root = logging.getLogger()
    if root.handlers:
        return  # don't clobber logging config of an embedding application
    handler = logging.StreamHandler()
    handler.setFormatter(JsonFormatter())
    root.addHandler(handler)
    root.setLevel(logging.INFO)


def run(path):
    _configure_logging()
    run_id = uuid.uuid4().hex[:8]
    started = time.monotonic()
    total = ingested = succeeded = skipped = failed = 0

    logger.info({"event": "run_start", "run_id": run_id, "path": path})

    with open(path) as f:
        for line_num, raw_line in enumerate(f, start=1):
            total += 1
            record_id = f"{path}:{line_num}"
            line = raw_line.strip()
            if not line:
                continue

            try:
                rec = json.loads(line)
            except json.JSONDecodeError as exc:
                failed += 1
                logger.error({
                    "event": "parse_error",
                    "run_id": run_id,
                    "record_id": record_id,
                    "error": str(exc),
                    "raw": line[:200],
                })
                continue

            ingested += 1
            record_id = str(rec.get("id", record_id))
            logger.info({"event": "record_received", "run_id": run_id, "record_id": record_id})

            try:
                out = enrich_record(rec, record_id=record_id, run_id=run_id)
            except Exception:
                failed += 1
                logger.exception({
                    "event": "enrich_failed",
                    "run_id": run_id,
                    "record_id": record_id,
                })
                continue

            if out is None:
                skipped += 1
                logger.info({"event": "record_skipped", "run_id": run_id, "record_id": record_id})
                continue

            try:
                save(out)
            except Exception:
                failed += 1
                logger.exception({
                    "event": "save_failed",
                    "run_id": run_id,
                    "record_id": record_id,
                })
                continue

            succeeded += 1
            logger.info({"event": "record_complete", "run_id": run_id, "record_id": record_id})

    logger.info({
        "event": "run_complete",
        "run_id": run_id,
        "path": path,
        "total_lines": total,
        "ingested": ingested,
        "succeeded": succeeded,
        "skipped": skipped,
        "failed": failed,
        "duration_s": round(time.monotonic() - started, 3),
    })


def save(rec):
    with open("out.jsonl", "a") as f:
        f.write(json.dumps(rec) + "\n")
```

**enrich.py**
```python
import logging

logger = logging.getLogger(__name__)

GEO = {"us": "US-EAST", "eu": "EU-WEST"}


def enrich_record(rec, record_id="unknown", run_id=None):
    region_code = rec.get("region")
    if region_code is None:
        logger.warning({
            "event": "missing_region",
            "run_id": run_id,
            "record_id": record_id,
        })
        return None

    region = GEO.get(region_code)
    if region is None:
        logger.warning({
            "event": "unknown_region",
            "run_id": run_id,
            "record_id": record_id,
            "region": region_code,
        })
        return None

    rec["zone"] = region
    logger.debug({
        "event": "enriched",
        "run_id": run_id,
        "record_id": record_id,
        "zone": region,
    })
    return rec
```

Notable fixes beyond logging: `rec["region"]` was changed to `rec.get("region")` since a missing key would otherwise raise `KeyError` and abort that record ungracefully instead of being logged as a clear, distinct event; and the file is now opened with a `with` block instead of a bare `open(path)` to avoid leaking file handles across thousands of runs.
