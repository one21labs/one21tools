```python
import logging
import requests
from requests.adapters import HTTPAdapter, Retry

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
logger = logging.getLogger(__name__)

session = requests.Session()
session.mount("https://", HTTPAdapter(max_retries=Retry(
    total=3, backoff_factor=0.5, status_forcelist=[429, 500, 502, 503, 504]
)))
```
Then replace `requests.get(...)` with `session.get(...)`, and wrap each call:
```python
logger.info("Fetching users from %s", BASE + "/users")
response = session.get(BASE + "/users")
logger.info("Got %s users (status %s)", len(response.json()), response.status_code)
```
