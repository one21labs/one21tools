# Sanitizing Customer Exports

A Claude Code skill that redacts personally identifiable information (PII) from customer CSV exports before sharing outside the company.

## Overview

This skill automatically processes customer CSV files to remove sensitive data while preserving the structure and non-sensitive content. It ensures compliance with data privacy requirements by systematically redacting PII across multiple dimensions.

## How It Works

The skill applies the following redaction rules:

### Column Removal
The following columns are completely dropped from the output:
- `email`
- `phone`
- `ssn`

### Column Transformation
- **name** column: Replaced with an opaque identifier (hash-based ID)

### Free-Text Scanning
Free-text columns (`notes`, `comments`) are scanned for patterns matching:
- Email addresses (e.g., user@example.com)
- Phone numbers (e.g., 555-123-4567, (555) 123-4567)

Detected substrings are masked with asterisks while preserving the overall text structure.

## File 2: sanitize.py

```python
#!/usr/bin/env python3
"""
Customer CSV export PII redaction utility.
Removes sensitive columns, replaces names with opaque IDs, and masks PII in free-text fields.
"""

import csv
import re
import hashlib
import sys
from pathlib import Path
from typing import Dict, Set, List, Tuple


EMAIL_PATTERN = re.compile(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b')
PHONE_PATTERN = re.compile(r'(\+?1[-.\s]?)?\(?\d{3}\)?[-.\s]?\d{3}[-.\s]?\d{4}')

COLUMNS_TO_DROP = {'email', 'phone', 'ssn'}
FREETEXT_COLUMNS = {'notes', 'comments'}


def hash_to_opaque_id(value: str) -> str:
    """Generate a deterministic opaque ID from a string."""
    hash_obj = hashlib.sha256(value.encode('utf-8'))
    return hash_obj.hexdigest()[:8]


def mask_emails(text: str) -> str:
    """Replace email addresses with masked versions."""
    def mask(match):
        email = match.group(0)
        parts = email.split('@')
        return '*' * len(parts[0]) + '@' + '*' * len(parts[1].split('.')[0]) + '.' + '*' * 3
    return EMAIL_PATTERN.sub(mask, text)


def mask_phones(text: str) -> str:
    """Replace phone numbers with masked versions."""
    def mask(match):
        return '*' * len(match.group(0))
    return PHONE_PATTERN.sub(mask, text)


def sanitize_row(row: Dict[str, str], headers: List[str], 
                 name_index: int, freetext_indices: Set[int]) -> Tuple[Dict[str, str], bool]:
    """
    Sanitize a single row.
    Returns: (sanitized_row, has_name_column)
    """
    sanitized = {}
    has_name = False
    
    for i, header in enumerate(headers):
        header_lower = header.lower()
        
        # Skip columns that should be dropped
        if header_lower in COLUMNS_TO_DROP:
            continue
        
        # Replace name with opaque ID
        if header_lower == 'name':
            has_name = True
            original_name = row.get(header, '')
            sanitized[header] = hash_to_opaque_id(original_name)
            continue
        
        # Scan and mask free-text columns
        if i in freetext_indices:
            text = row.get(header, '')
            text = mask_emails(text)
            text = mask_phones(text)
            sanitized[header] = text
        else:
            # Keep other columns as-is
            sanitized[header] = row.get(header, '')
    
    return sanitized, has_name


def process_csv(input_path: str) -> None:
    """Process a single CSV file and create redacted version."""
    input_file = Path(input_path)
    
    if not input_file.exists():
        print(f"Error: File not found: {input_path}")
        return
    
    if not input_file.suffix.lower() == '.csv':
        print(f"Error: File is not a CSV: {input_path}")
        return
    
    output_path = input_file.parent / f"{input_file.stem}.redacted.csv"
    
    try:
        with open(input_file, 'r', encoding='utf-8', newline='') as infile:
            reader = csv.DictReader(infile)
            
            if reader.fieldnames is None:
                print(f"Error: CSV file has no headers: {input_path}")
                return
            
            headers = reader.fieldnames
            headers_lower = [h.lower() for h in headers]
            
            # Find indices for name and free-text columns
            name_index = None
            freetext_indices = set()
            
            for i, header in enumerate(headers_lower):
                if header == 'name':
                    name_index = i
                if header in FREETEXT_COLUMNS:
                    freetext_indices.add(i)
            
            # Build output headers (excluding dropped columns)
            output_headers = [h for h, hl in zip(headers, headers_lower) 
                             if hl not in COLUMNS_TO_DROP]
            
            # Process rows
            sanitized_rows = []
            has_name_column = False
            
            for row in reader:
                sanitized_row, has_name = sanitize_row(row, headers, name_index, freetext_indices)
                sanitized_rows.append(sanitized_row)
                if has_name:
                    has_name_column = True
            
            # Write output
            with open(output_path, 'w', encoding='utf-8', newline='') as outfile:
                writer = csv.DictWriter(outfile, fieldnames=output_headers)
                writer.writeheader()
                writer.writerows(sanitized_rows)
            
            print(f"✓ Sanitized: {input_path} → {output_path.name}")
            print(f"  Dropped columns: {', '.join(sorted(COLUMNS_TO_DROP & set(headers_lower)))}")
            if has_name_column:
                print(f"  Name column replaced with opaque IDs")
            if freetext_indices:
                freetext_cols = [headers[i].lower() for i in sorted(freetext_indices)]
                print(f"  Scanned for PII in: {', '.join(freetext_cols)}")
    
    except Exception as e:
        print(f"Error processing {input_path}: {e}")


def main():
    """Process all CSV files provided as arguments."""
    if len(sys.argv) < 2:
        print("Usage: python sanitize.py <csv_file> [<csv_file> ...]")
        print("\nThis tool redacts PII from customer CSV exports:")
        print("  - Drops: email, phone, ssn columns")
        print("  - Replaces: name with opaque ID")
        print("  - Masks in notes/comments: emails and phone numbers")
        print("\nOutput is saved as <filename>.redacted.csv")
        sys.exit(1)
    
    files = sys.argv[1:]
    
    for file_path in files:
        process_csv(file_path)
    
    if len(files) > 1:
        print(f"\nProcessed {len(files)} files")


if __name__ == '__main__':
    main()
```
