## scripts/fetch_users.py

```python
import logging
import requests

logger = logging.getLogger(__name__)

BASE = "https://api.newhost.io/v2"
MAX_RETRIES = 3

def fetch_users():
    for attempt in range(1, MAX_RETRIES + 1):
        try:
            logger.info("Fetching users (attempt %d/%d)", attempt, MAX_RETRIES)
            response = requests.get(BASE + "/users")
            response.raise_for_status()
            return response.json()
        except requests.exceptions.RequestException as e:
            logger.warning("fetch_users attempt %d failed: %s", attempt, e)
            if attempt == MAX_RETRIES:
                logger.error("fetch_users failed after %d attempts", MAX_RETRIES)
                raise
```
